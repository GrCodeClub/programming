      Συμβολοσειρές

      Μετατροπή σε άλλους τύπους (stdlib.h)

      Μετατρέπει μια συμβολοσειρά σε ακέραιο.

      int atoi(const char *str);

      Παράδειγμα

      #include <stdlib.h>               

int main() {
    const char *numberStr = "42";
    // Μετατροπή της συμβολοσειράς "42" σε ακέραιο 42
    int number = atoi(numberStr);
    printf("The number is: %d\n", number);
    return 0;
    }

      Μετατρέπει μια συμβολοσειρά σε ακέραιο μεγάλου μεγέθους.

      long atol(const char *str);

      Παράδειγμα

      #include <stdio.h>
#include  <stdlib.h>

int main() {
    const char *numberStr = "1234567890";
    // Μετατροπή της συμβολοσειράς "1234567890" σε long int 
    long number = atol(numberStr);                       
    printf("The number is: %ld\n", number);             
    return 0;
    }

      Μετατρέπει μια συμβολοσειρά σε δεκαδικό αριθμό.

      double atof(const char *str);

      Παράδειγμα

      #include <stdio.h>
#include <stdlib.h>
                
int main() {
    const char *numberStr = "3.14159";
    // Μετατροπή της συμβολοσειράς "3.14159" σε κινητή υποδιαστολή
    double number = atof(numberStr);       
    printf("The number is: %lf\n", number);             
    return 0;
    }

      Συναρτήσεις διαχείρισης συμβολοσειρών (string.h)

      Η συνάρτηση strcat (string concatenation) χρησιμοποιείται για να συνενώσει (να ενώσει) δύο συμβολοσειρές. Αυτή η συνάρτηση προσθέτει το περιεχόμενο της δεύτερης συμβολοσειράς (src) στο τέλος της πρώτης συμβολοσειράς (dest), αλλάζοντας έτσι το περιεχόμενο της πρώτης συμβολοσειράς.

      char* strcat(char* dest, const char* src);

      Παράδειγμα

      #include <stdio.h>
#include  <string.h>

int main() {
    char destination[50] = "Hello, ";
    const char source[] = "World!";              
    // Συνένωση των συμβολοσειρών           
    strcat(destination, source);                       
    printf("Ενωμένη συμβολοσειρά: %s\n", destination);                        
    return 0;
}

      Συγκρίνει τις συμβολοσειρές string1 and string2. Επιστρέφει 0 όταν είναι ίσες.

      int strcmp(const char* string1, const char* string2);

      Παράδειγμα

      #include <stdio.h>
#include  <string.h>

int main() {
    const char *str1 = "apple";
    const char *str2 = "banana";
    int result = strcmp(str1, str2);

    if (result < 0) {
        printf("'%s'είναι μικρότερο απο'%s'\n", str1, str2);
    } else if (result > 0) {
        printf("'%s' είναι μεγαλύτερη απο'%s'\n", str1, str2);
    } else {
        printf("'%s' είναι ίσο με '%s'\n", str1, str2);
    }
    return 0;
}

      Αντιγράφει το περιεχόμενο της src στη dest. Επιστρέφει την dest.

      char* strcpy (char* dest, const char* src);

      Παράδειγμα

      #include <stdio.h>
#include  <string.h>

int main() {
    const char *source = "Hello, World!";
    char destination[50];
    // Αντιγραφή του περιεχομένου source στo destination
    strcpy(destination, source); 
    printf("Source: %s\n", source);
    printf("Destination: %s\n", destination);
    return 0;
}

      Επιστρέφει το μήκος της συμβολοσειράς.

      size_t strlen (const char* string);

      Παράδειγμα

      #include <stdio.h>
#include  <string.h>

int main() {
    const char *str = "Hello, World!";
    // Υπολογισμός μήκους της συμβολοσειράς
    size_t length = strlen(str); 
    
    printf("The length of the string is: %zu\n", length);
    
    return 0;
}
