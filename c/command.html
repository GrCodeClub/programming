      Σε αυτό το παράδειγμα, ζητάμε από τον χρήστη να εισάγει έναν ακέραιο αριθμό. Στη συνέχεια, χρησιμοποιούμε μια δομή επανάληψης for για να εκτυπώσουμε όλους τους αριθμούς από το 1 μέχρι και τον αριθμό που εισήγαγε ο χρήστης, χρησιμοποιώντας την cout.

      for.cpp
      #include  <iostream >
using namespace std;
   
int main() {
    int n;
    cout << "Παρακαλώ εισάγετε έναν ακέραιο αριθμό: ";
    cin >> n;
   
    // Εκτύπωση των αριθμών από το 1 μέχρι το n
    cout << "Οι αριθμοί από το 1 μέχρι το " << n << " είναι: ";
    for (int i = 1; i <= n; i++) {
        cout << i << " "; // Εκτύπωση του αριθμού i
    }
    cout << endl;
   
    return 0;
}

      Σε αυτό το παράδειγμα, ζητάμε από τον χρήστη να εισάγει έναν βαθμό. Στη συνέχεια, χρησιμοποιούμε τη δομή ελέγχου if-else if-else για να αξιολογήσουμε την τιμή του βαθμού και να εκτυπώσουμε ανάλογο μήνυμα ανάλογα με το επίπεδο απόδοσης.

      if.cpp
      #include  <iostream >
using namespace std;
   
int main() {
    int score;
    cout << "Παρακαλώ εισάγετε τον βαθμό σας: ";
    cin >> score;
   
    if (score >= 90) {
        cout << "Εξαιρετική επίδοση! Βαθμός Α" << endl;
    } else if (score >= 80) {
        cout << "Πολύ καλή επίδοση! Βαθμός Β" << endl;
    } else if (score >= 70) {
        cout << "Καλή επίδοση! Βαθμός Γ" << endl;
    } else if (score >= 60) {
        cout << "Μέτρια επίδοση! Βαθμός Δ" << endl;
    } else {
        cout << "Αδικαιολόγητη επίδοση! Βαθμός Ε" << endl;
    }
    return 0;
}

      Σε αυτό το παράδειγμα, ο χρήστης καλείται να εισαγάγει έναν αριθμό που αντιστοιχεί σε μια ημέρα της εβδομάδας. Η δομή ελέγχου switch ελέγχει την τιμή της μεταβλητής day και εκτελεί τις αντίστοιχες ενέργειες για κάθε περίπτωση (case).

      switch.cpp
      #include  <iostream >
using namespace std;
   
int main() {
    int day;
    cout << "Παρακαλώ εισάγετε έναν αριθμό για την ημέρα της εβδομάδας (1-7): ";
    cin >> day;
   
    switch (day) {
        case 1:
            cout << "Δευτέρα" << endl;
            break;
        case 2:
            cout << "Τρίτη" << endl;
            break;
        case 3:
            cout << "Τετάρτη" << endl;
            break;
        case 4:
            cout << "Πέμπτη" << endl;
            break;
        case 5:
            cout << "Παρασκευή" << endl;
            break;
        case 6:
            cout << "Σάββατο" << endl;
            break;
        case 7:
            cout << "Κυριακή" << endl;
            break;
        default:
            cout << "Μη έγκυρη είσοδος!" << endl;
            break;
    }
    return 0;
}
